# GitHub Setup Commands (run after installing Git)

# 1. Initialize repository
git init

# 2. Add all files
git add .

# 3. Create initial commit
git commit -m "Initial commit: AppSettingUpdater v2.0 - Enterprise PowerShell module for Azure Web App deployments with parallel processing"

# 4. Add remote repository (replace with your GitHub repo URL)
git remote add origin https://github.com/yourusername/AppSettingUpdater.git

# 5. Push to GitHub
git branch -M main
git push -u origin main

# Alternative: If you want to create the repo via GitHub CLI
gh repo create AppSettingUpdater --public --description "Enterprise PowerShell module for automated Azure Web App deployments with zero-downtime slot swapping and parallel processing"
